  <action>
    1. Create `schemas/blog-post.ts` -- the Sanity document schema for blog posts:
       ```ts
       import { defineType, defineField } from 'sanity'

       export const blogPost = defineType({
         name: 'blogPost',
         title: 'Blog Post',
         type: 'document',
         fields: [
           defineField({
             name: 'title',
             title: 'Title',
             type: 'string',
             validation: (rule) => rule.required(),
           }),
           defineField({
             name: 'slug',
             title: 'Slug',
             type: 'slug',
             options: { source: 'title', maxLength: 96 },
             validation: (rule) => rule.required(),
           }),
           defineField({
             name: 'author',
             title: 'Author',
             type: 'string',
             validation: (rule) => rule.required(),
           }),
           defineField({
             name: 'pubDate',
             title: 'Publication Date',
             type: 'datetime',
             validation: (rule) => rule.required(),
           }),
           defineField({
             name: 'category',
             title: 'Category',
             type: 'string',
             options: {
               list: [
                 { title: 'AI Search & Visibility', value: 'ai-search-visibility' },
                 { title: 'Relevance Engineering', value: 'relevance-engineering' },
                 { title: 'Agentic Commerce', value: 'agentic-commerce' },
                 { title: 'Prompt Intelligence', value: 'prompt-intelligence' },
                 { title: 'Industry Insights', value: 'industry-insights' },
               ],
             },
           }),
           defineField({
             name: 'featuredImage',
             title: 'Featured Image',
             type: 'image',
             options: { hotspot: true },
           }),
           defineField({
             name: 'seoDescription',
             title: 'SEO Description',
             type: 'text',
             rows: 3,
             description: 'Used for meta description and BlogPosting schema. Keep under 160 characters.',
           }),
           defineField({
             name: 'content',
             title: 'Content',
             type: 'array',
             of: [
               {
                 type: 'block',
                 styles: [
                   { title: 'Normal', value: 'normal' },
                   { title: 'H2', value: 'h2' },
                   { title: 'H3', value: 'h3' },
                   { title: 'Quote', value: 'blockquote' },
                 ],
                 marks: {
                   decorators: [
                     { title: 'Bold', value: 'strong' },
                     { title: 'Italic', value: 'em' },
                   ],
                   annotations: [
                     {
                       name: 'link',
                       type: 'object',
                       title: 'Link',
                       fields: [
                         { name: 'href', type: 'url', title: 'URL' },
                       ],
                     },
                   ],
                 },
               },
             ],
           }),
         ],
         orderings: [
           {
             title: 'Publication Date (Newest)',
             name: 'pubDateDesc',
             by: [{ field: 'pubDate', direction: 'desc' }],
           },
         ],
         preview: {
           select: {
             title: 'title',
             author: 'author',
             media: 'featuredImage',
           },
           prepare({ title, author, media }) {
             return {
               title,
               subtitle: author,
               media,
             }
           },
         },
       })
       ```

       IMPORTANT per locked decisions:
       - Author is simple text field (NOT a reference to an Author document)
       - NO excerpt field, NO reading time field
       - Featured image only (no inline images in content blocks)
       - Categories are predefined (Claude's discretion on names -- chosen to match consultancy pillars)
       - Added seoDescription field to solve BlogPostingSchema's required `description` prop (existing schema requires it for Google rich results)

         author,
         pubDate,
         category,
         seoDescription,
         featuredImage,
       }`

       export const getPostBySlugQuery = `*[_type == "blogPost" && slug.current == $slug][0] {
         _id,
         title,
         slug,
         author,
         pubDate,
         category,
         seoDescription,
         featuredImage,
         content,
       }`
       ```

    5. Create `src/lib/sanity/image.ts`:

         projectId: 'pbui2f8s',
         dataset: 'production',
       })

       export const urlFor = (source: any) => builder.image(source)
       ```
       NOTE: Use `createImageUrlBuilder` named export -- NOT default import. This is a v2 breaking change in @sanity/image-url.
